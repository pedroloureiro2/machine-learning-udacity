def preprocess_features(X):
    
    output = pd.DataFrame(index = X.index)    
    for col, col_data in X.iteritems():
        # Se o tipo de dado for categrico, converta-o para uma varivel dummy
        if col_data.dtype == object:        
             col_data = pd.get_dummies(col_data, prefix = col)               
        
        # Rena as colunas revisadas
        output = output.join(col_data)
    
    return output

from time import time
from sklearn.metrics import f1_score

def train_classifier(clf, X_train, y_train):
    ''' Ajusta um classificador para os dados de treinamento. '''
    
    # Inicia o relgio, treina o classificador e, ento, para o relgio
    start = time()
    clf.fit(X_train, y_train)
    end = time()
    
    # Imprime os resultados
    print("O modelo foi treinado em {:.4f} segundos".format(end - start))

    
def predict_labels(clf, features, target):
    ''' Faz uma estimativa utilizando um classificador ajustado baseado na pontuao F1. '''
    
    # Inicia o relgio, faz estimativas e, ento, o relgio para
    start = time()
    y_pred = clf.predict(features)
    end = time()    
    # Imprime os resultados de retorno
    print("As previses foram feitas em {:.4f} segundos.".format(end - start))
    return f1_score(target.values, y_pred)


def train_predict(clf, X_train, y_train, X_test, y_test):
    ''' Treina e faz estimativas utilizando um classificador baseado na pontuao do F1. '''
    
    # Indica o tamanho do classificador e do conjunto de treinamento
    print("")
    print("Treinando um {} com {} pontos de treinamento. . .".format(clf.__class__.__name__, len(X_train)))
    
    # Treina o classificador
    train_classifier(clf, X_train, y_train)
    
    # Imprime os resultados das estimativas de ambos treinamento e teste
    print("Pontuao F1 para o conjunto de treino: {:.4f}.".format(predict_labels(clf, X_train, y_train)))
    print("Pontuao F1 para o conjunto de teste: {:.4f}.".format(predict_labels(clf, X_test, y_test)))
